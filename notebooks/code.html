 26def f(T, L, a=1/2, th=1):
 27    '''production technology 
 28       $$f(T, L) = \\theta \cdot T^{1-\\alpha}L^{\\alpha}$$ 
 29       '''
 30    return th * T**(1-a) * L**a
 31
 32def mple(te, le, a=1/2, th=1, tlbar=Tbar/Lbar):
 33    '''marginal product of Labor on enclosed land can be written
 34       $$MPL(t_e, l_e) = \\alpha \\cdot \\frac{f(t_e, l_e)}{l_e} \\bar l^\\alpha$$ 
 35       since it is a Cobb Douglas $$MPL = \\alpha \\cdot APL$$. '''
 36    return a* f(te,le,a,th)/le  * tlbar**(1-a)
 37
 38def aple(te, le, a=1/2, th=1, tlbar=Tbar/Lbar):
 39    '''Average product of Labor 
 40    $$APL(t_e, l_e) = \\frac{f(T_e, L_e)}{L_e} =  \\frac{f(t_e, l_e)}{l_e} \cdot \\bar t^{1-\\alpha}$$ 
 41    '''
 42    return f(te,le,a,th)/le  * tlbar**(1-a)
 43
 44def mpte(te, le, a=1/2, th=1, tlbar=Tbar/Lbar):
 45    '''marginal product of Land on enclosed land'''
 46    return (1-a)* f(te,le,a,th)/te  * tlbar**(-a)
 47
 48def mplu(te, le, a=1/2, th=1, tlbar=Tbar/Lbar):
 49    '''marginal product of Labor on unenclosed land
 50       same tech but useful to have other name'''
 51    return mple(te, le, a, th, tlbar)
 52
 53def aplu(te, le, a=1/2, th=1, tlbar=Tbar/Lbar):
 54    '''average product of Labor on unenclosed land'''
 55    return aple(te, le, a, th, tlbar)

 57def Lambda(th, alp, mu):
 58    ''' Key parameter for expressions. Can return either private or planners Lambda:
 59    $$\\mu = 0 \\rightarrow \\Lambda = (\\alpha \\theta)^\\frac{1}{1-\\alpha}$$
 60    $$\\mu = 1 \\rightarrow \\Lambda_o = \\theta^\\frac{1}{1-\\alpha}$$
 61    '''
 62    return ( (alp*th)/(1-mu+alp*mu) )**(1/(1-alp))

 65def req(te, th=1, alp=1/2, ltbar=1, mu=0):
 66    '''Decentralized Equilibrium rental
 67       $$r(t_e) =  \\theta f_T(t_e, l_e(t_e)) \\cdot \\bar t^\\alpha$$ 
 68       $$r(t_e) =  \\frac{(1-\\alpha) \\theta  \\Lambda}{(1+(\\Lambda-1)t_e)^\\alpha}  \\cdot \\bar t^\\alpha$$ 
 69
 70    '''
 71    lam = Lambda(th, alp, mu)
 72    return (1-alp)*th * lam**alp * (1+(lam-1)*te)**(-alp) * (ltbar)**(alp)
 73
 74
 75def weq(te, th=1, alp=1/2, tlbar=1, mu =0):
 76    '''Decentralized Equilibrium wage'''
 77    lam = Lambda(th, alp, mu)
 78    return (1+(lam-1)*te)**(1-alp) * (tlbar)**(1-alp)
 79
 80def leo(te, th, alp):
 81    '''optimal labor allocation (from MPLe = MPLu) given enclosed land share te'''
 82    lam = th**(1/(1-alp))
 83    return (lam*te)/(1+lam*te-te)
 84
 85def le(te, th, alp, mu):
 86    '''private eqn labor share on enclosed for given te when 
 87       (1-mu*alp*mu)*APL=MPL  
 88       mu = 0:   APLc = MPLe,   le* in paper
 89       mu = 1:   MPLc = MPLe,   leo in paper
 90       mu in (0,1)   in between partly secure
 91       '''
 92    lam = Lambda(th, alp, mu)
 93    return (lam*te)/(1+(lam-1)*te)
 94
 95def totalq(te, th, alp, lbar, mu):
 96    '''total output in the economy given te and mu.
 97       Note costs of enclosure are not subtracted.'''
 98    leq = le(te, th, alp, mu)
 99    return ( th * f(te, leq, alp, th) + f(1-te, 1-leq, alp, 1) ) * lbar**alp

265def z(te, th, alp, lbar):
266    '''output per unit land net of enclosure cost
267       $$z(t_e) = \\bar l^\\alpha \\left(1+(\\Lambda_o-1)t_e\\right)^{1-\\alpha}$$ '''
268    lam = th**(1/(1-alp))
269    return lbar**alp * (1+(lam-1)*te)**(1-alp)

271def zpv(te, th, alp, lbar):
272    '''output per unit land net of enclosure cost  NEED TO ADJUST:
273       $$z_d(t_e)= \\bar l^\\alpha \\cdot \\frac{ 1+(\\frac{\\Lambda}{\\alpha}-1)t_e}{(1+(\\Lambda-1)t_e)^\\alpha}$$
274  '''
275    lam = (alp*th)**(1/(1-alp))
276    return lbar**alp * (th*te*lam**alp +(1-te))/(1+(lam-1)*te)**alp

280def zprime(te, th, alp, lbar):
281    '''derivative of planner's z(t_e) function
282       $$z(t_e) = \\bar l^\\alpha \\cdot (1-\\alpha)(\\Lambda_o -1) \\left(1+(\\Lambda_o-1)t_e \\right)^{-\\alpha}$$
283       '''
284    lam = th**(1/(1-alp))
285    return  (1-alp)*(lam-1)*lbar**alp  * (1+(lam-1)*te)**(-alp)

288def teopt(th, alp, c, lbar):
289    '''Planner enclosure rate. If partial then
290    $$t_e^o =  \\frac{\\bar l}{(\\Lambda_o - 1)} 
291    \\left [   \\frac{(1-\\alpha)(\\Lambda_o - 1)}{c} 
292    \\right ]^\\frac{1}{\\alpha} - \\frac{1}{(\\Lambda_o - 1)}
293    $$'''
294    lam = th**(1/(1-alp))
295    zprime = lambda te : (1-alp)*(lam-1)*lbar**alp  * (1+(lam-1)*te)**(-alp) 
296    if zprime(0)<c:
297        teopt = 0
298    elif zprime(1)>c:
299        teopt = 1
300    else:
301        teopt =  ( lbar * (  ((1-alp)*(lam-1))/c)**(1/alp)  - 1)/(lam-1)
302
303    return teopt

306def tepvt(th, alp, c, lbar, mu):
307    '''Private enclosure rate
308        req(te)= rental rate 
309        r(0)<c  : no enclosure 
310        r(1)>c  : full enclosure
311        r(0)>c and r(1)<c : partial enclosure
312           then solve for teopt from foc
313
314        '''
315    thresh = (1-mu+alp*mu)/alp    
316    lam = Lambda(th, alp, mu)
317    r0 = req(0, th, alp, lbar)
318    r1 = req(1, th, alp, lbar)
319    if th<thresh:
320        if r0>=c:
321            tep = 1
322        elif r1<c:
323            tep = 0
324        else:
325            tep = lbar * (lam/(lam-1)) * (th*(1-alp)/c )**(1/alp) - (1/(lam-1))
326    
327    elif th>= thresh:  
328        if r1>=c:
329            tep = 1
330        elif r0<c:
331            tep = 0
332        else:
333            tep = lbar * ( lam/(lam-1)) * (th*(1-alp)/c )**(1/alp) - (1/(lam-1))
334
335    return tep

338def tepvt_g(th, alp, c, lbar, mu):
339    '''Private enclosure rate (global game refinement)
340        just like pvtpart() but adjust for global game
341        If theta < theta_hi then global game refinement says enclose fully if
342        tep (from pvtpart) <= 0.5 otherwise no enclosre.
343        '''
344    thresh = (1-mu+alp*mu)/alp    
345    tep = tepvt(th,alp,c, lbar, mu=0)
346    
347    tepg = tep
348    if (tep==1) or (tep==0):
349        tepg = tep
350    elif (th < thresh):
351        if (tep > 0.5):
352            tepg = 0
353        elif (tep <= 0.5):
354            tepg = 1
355
356    return tepg

360def dwl(th, alp, c, lbar):
361    '''
362    Returns DWL at each paramter
363    '''
364    teo= teopt(th, alp, c, lbar)
365    tep = tepvt(th,alp,c, lbar, mu=0)

369    zg = zpv(teg, th, alp, lbar) - c*teg
370    return  zo-zg

432def socpart(c = 1, alp= 2/3, mu =0, soc_opt= True, cond_opt=True, pv_opt=False, logpop=True):
433    '''Plots loci determining parameter partitions corresponding to 
434        Social (and Conditional Social) Optimum
435        None, Full, or Partial Enclosure zones
436        Option: to log plot or not
437        '''
438    start, finish  = 1.1, 2.1   # range that will be plotted
439    the_1 = np.arange(start, finish, .01)
440    cv = 1 / alp                          # high TFP gain threshold
441    the_lo = np.arange(start, cv, .01)
442    the_hi = np.arange(cv, finish, .01)
443    
444    ## Social Optima
445    lamO = the_1**(1/(1-alp))
446
447    lo0 = ( c / ( (lamO - 1)*(1-alp) )  ) **(1/alp)
448    lo1 = lamO * lo0      
449
450    ##### Conditional Optima:  we need separate plot ranges, each side of cv = theta_hat
451    lam_hi =  Lambda(th = the_hi, alp= alp, mu = mu)
